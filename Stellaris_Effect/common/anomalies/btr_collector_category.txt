btr_collector_category_1 = {
	picture = GFX_evt_warm_barren
	level = 3

	spawn_chance = {
		modifier = {
			has_global_flag = btr_collectors_enabled
			OR = {
				is_planet_class = pc_barren
				is_planet_class = pc_barren_cold
			}
			add = 6
		}
	}
	
	on_spawn = {
		FROM = {
			owner = {
				set_country_flag = btr_collector_abducted_world_found
			}
		}
	}

	on_success = {
		1 = btr_collector.202
		1 = btr_collector.203
		1 = {
			anomaly_event = btr_collector.204
			modifier = {
				factor = 0
				owner = {
					has_ethic = ethic_gestalt_consciousness
				}
			}
		}		
	}
}
