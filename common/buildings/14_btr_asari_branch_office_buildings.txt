# These buildings represent entering a partnership with a local business, giving the host planet specialist jobs while increasing the diplo weight and value of all branch offices for the megacorp
btr_asari_branch_illium_inv_entertainment = {
	base_buildtime = 10 #@b1_time
	base_cap_amount = 1

	branch_office_building = yes

	icon = btr_asari_branch_illium_building

	potential = {
		has_branch_office = yes
		branch_office_owner = { has_valid_civic = btr_asari_civic_illium }
		NOR = {
			has_building = btr_asari_branch_illium_inv_consumer
			has_building = btr_asari_branch_illium_inv_alloys
		}
	}

	allow = {
		custom_tooltip = {
			fail_text = "requires_building_major_capital"
			OR = {
				has_building = building_capital
				has_building = building_major_capital
				has_building = building_system_capital
				has_building = building_imperial_capital
			}
		}
	}

	country_modifier = {
		diplo_weight_pops_mult = 0.10
		branch_office_value_mult = 0.05
	}

	planet_modifier = {
		trade_value_add = 25
		job_entertainer_add = 1
		job_entertainer_per_pop = 0.025
		job_manager_add = 1
	}

	resources = {
		category = planet_branch_office_buildings
		cost = {
			influence = 200
		}
		upkeep = {
			influence = 1
		}
	}

	ai_weight = {
		weight = 50
		modifier = {
			factor = 0
			branch_office_owner = {
				has_monthly_income = {
					resource = influence
					value < 3
				}
			}
		}
		modifier = {
			factor = 2
			OR = {
				has_building = building_holo_theatres
				has_building = building_hyper_entertainment_forum
			}
		}
	}
}
