

### Salarian Union ###

btr_decision_saluni_mark_for_giving = {
	owned_planets_only = yes
	icon = decision_resources

	effect = {
		set_planet_flag = btr_saluni_gift_mark
	}

	potential = {
		owner = { has_technology = btr_tech_saluni_exo_arch }
		NOT = { has_planet_flag = btr_saluni_gift_mark }
	}
}

btr_decision_saluni_unmark_for_giving = {
	owned_planets_only = yes
	icon = decision_resources

	effect = {
		remove_planet_flag = btr_saluni_gift_mark
	}

	potential = {
		 has_planet_flag = btr_saluni_gift_mark 
	}
}
