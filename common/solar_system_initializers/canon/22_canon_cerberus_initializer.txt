		has_ring = no
		starting_planet = yes

		deposit_blockers = none
		modifiers = none

		init_effect = {
			random_country = {
				limit = { has_country_flag = cerberus }
				save_global_event_target_as = cerberus
			}
			prevent_anomaly = yes
			set_global_flag = cronos_station_spawned

			if = {

				    class = HUM

				    homeworld = THIS 
				    traits = { 
						trait="btr_human_flexible"
						trait="btr_human_multicultural"
						trait="trait_wasteful"
						trait="trait_quick_learners"
					} 
				}
				last_created_species = { save_global_event_target_as = humanSpecies }
				create_country = {
					name = "NAME_Cerberus"
					ship_prefix = "PRESCRIPTED_ship_prefix_cerberus"
					type = default
					origin = "origin_cerberus"
					ignore_initial_colony_error = yes
					civics = { civic = "civic_efficient_bureaucracy" civic = "civic_functional_architecture" }
					authority = auth_democratic
					name_list = "btr_cerberus"
					ethos = { ethic = "ethic_fanatic_xenophobe" ethic = "ethic_militarist" }
					species = event_target:humanSpecies
					flag = {
						icon={
							category="masseffect"
							file="Cerberus 2.dds"
						}
						background={
							category="backgrounds"
							file="v.dds"
						}
						colors={
							"orange"
							"black"
							"null"
							"null"
						}
					}
					effect = {
						set_graphical_culture = reptilian_01
						set_country_flag = cerberus
						set_country_flag = init_spawned
						set_country_flag = custom_start_screen
						set_country_flag = canon_map
						save_global_event_target_as = cerberus
					}
				}
			}	
			set_capital = yes
			random_country = {
				limit = { has_country_flag = cerberus }
				save_global_event_target_as = cerberus
				species = { save_global_event_target_as = humanSpecies }
			}
			give_technology = {
				tech = tech_hydroponics
				message = no
			}
			set_planet_flag = megastructure
			set_planet_flag = habitat
			set_owner = event_target:cerberus
			generate_btr_start_deposits_and_blockers = yes
			generate_btr_cronos_buildings = yes
			generate_cronos_pops = yes
			add_deposit = d_hydroponics_bays
			add_deposit = d_cronos_science_labs
			add_deposit = d_cronos_reactor
			set_name = "Cronos Station"
		}
	}	
}
