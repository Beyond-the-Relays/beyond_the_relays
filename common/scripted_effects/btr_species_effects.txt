
### Illuminated Primacy ###
ensure_drell_exist = { 
	if = {
		limit = { NOT = { exists = event_target:drellSpecies } }
		create_species = {
			name = "Drell"
			plural = "Drells"
			adjective = "Drell"
			class = DRE
			portrait = drell
			namelist = BTR_drell
			traits = { 
				trait="trait_nonadaptive"
		        trait="trait_strong"
		        trait="trait_quick_learners"
		        trait="trait_resilient"
		        trait="trait_nomadic"
				ideal_planet_class = "pc_desert" 
			}
			effect = { save_global_event_target_as = drellSpecies }
		}
	}
	[[homeworld]
		$homeworld$ = { save_event_target_as = tempHomeworld }
		event_target:drellSpecies = { set_species_homeworld = event_target:tempHomeworld } 
	]
}
