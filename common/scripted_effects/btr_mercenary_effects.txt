		class = admiral
		species = event_target:mercenary_species
		name = random
		event_leader = yes
		skill = 3
		leader_age_min = 25
		leader_age_max = 45
		traits = {
			trait = leader_trait_mercenary_warrior
			trait = random_trait
		}
	}
	last_created_leader = {
		set_leader_flag = mercenary_merc_leader
	}
}

		limit = { years_passed < 40 }

			count = 6

				prefix = no
				graphical_culture = "pirate_01"
			}

			count = 4

				name = random
