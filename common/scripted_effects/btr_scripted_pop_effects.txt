#### Create Specific Species Pops

# Asari
generate_asariPop = { 
  if = {
    limit = { NOT = { exists = event_target:asariSpecies } }
    create_species = { 
      name = "Asari"
      class = "MECASA"
      portrait = mec_asari
      homeworld = THIS 
      traits = { 
		trait = "mec_asari_trait_core"
		trait = "mec_asari_trait_biotic"
		trait = "trait_charismatic"
		trait = "mec_asari_trait_sophisticated"
        ideal_planet_class = "pc_continental"
      }
    }
    last_created_species = { save_global_event_target_as = asariSpecies }
  }
  create_pop = {
    species = event_target:asariSpecies
    ethos = owner
  }
}

# Starting Pops
generate_owner_main_pop = {
	create_pop = {
		species = owner_main_species
		ethos = owner 
	  }
	} 

generate_major_capitol_starting_pops = {
	while = {
		count = 26
		generate_owner_main_pop = yes
	 }
}

generate_minor_empire_capitol_pops = {
	while = {	
		count = 20
		generate_owner_main_pop = yes
 }
}

	 }
}
