
ensure_country_noveria_scientist_exists = { 
	random_owned_pop_species = {
		limit = {
			OR = {
				has_citizenship_type = { country = prev type = citizenship_full }
			}
			is_sapient = yes
		}
		save_event_target_as = noveria_scientistSpecies
	}
	if = {
		limit = { NOT = { any_owned_leader = { has_leader_flag = $target$ } } }	
		create_leader = {
			class = scientist
			species = event_target:noveria_scientistSpecies
			name = random
			skill = 1
			traits = { trait = leader_trait_adaptable }
			event_leader = yes
			effect = {
				save_event_target_as = $target$
				set_leader_flag = $target$
				set_leader_flag = $leaderType|civilian$
				set_leader_flag = $leaderFlag1|countryAdministrator$
			}
		}
	}
	else = {
		random_owned_leader = {
			limit = { has_leader_flag = $target$ }
			save_event_target_as = $target$
		}
	}
}




ensure_noveria_scientist_exists = {
	if = {
		limit = { 
			NOT = { 
				exists = event_target:noveria_scientist@root
			} 
		}
		create_leader = { 
			name = random 
			species = root.species 
			class = scientist
			effect = { 
				save_global_event_target_as = noveria_scientist@root
				save_event_target_as = noveria_scientist
				set_leader_flag = noveria_scientist
				recruitable = no
			}
		}
	}
	else = {
		event_target:noveria_scientist@root = { save_event_target_as = noveria_scientist }
	}
}
