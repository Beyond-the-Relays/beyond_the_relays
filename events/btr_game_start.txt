namespace = btr_game_start

### Faction initializer event
country_event = {
	id = btr_game_start.1
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		### Asari Republics Faction Setup
		if = {
			limit = {
				has_origin = btr_origin_asrep
			}
			enable_faction_of_type = xenoist
			break = yes
		}
	}
}
