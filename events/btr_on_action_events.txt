# Non-asari leaders have a chance of getting educated in Thessia trait
country_event = {
	id = btr_asrep.02
	is_triggered_only = yes
	hide_window = yes

	trigger = {
			AND = {
				NOT = {
					from = {
						species = { has_trait = mec_asari_trait_core }
					}
				}
				has_global_flag = btr_asari_rep_university_open_doors
			}
	}

	immediate = {
		from = {
			if = {
				limit = { leader_class = scientist }
				add_trait = btr_leader_trait_thessian_grad
				add_skill = 1
			}
		}
	}
}

# Species leaders get the respective leader trait
country_event = {
	id = btr_asrep.03
	is_triggered_only = yes
	hide_window = yes

	trigger = {
		always = yes
	}

	immediate = {
		if = {
			limit = { from = { species = { has_trait = mec_asari_trait_core } } }
			from = {
				add_trait = btr_leader_trait_asari
				add_skill = 2
			}
		}
		else_if = {
			limit = { 
				from = { 
					species = { has_trait = btr_turian_trait_core }
					OR = {
						leader_class = general
						leader_class = admiral
					}
				} 
			}
			from = {
				add_trait = btr_leader_trait_extensive_training
			}
		}
		else_if = {
			limit = { 
				from = { 
					species = { has_trait = btr_geth_trait_core }
				} 
			}
			from = {
				add_trait = btr_leader_trait_internal_consensus
			}
		}
		else_if = {
			limit = { 
				from = {
					species = { has_trait = btr_human_core }
				}
			}
			from = {
				add_trait = btr_leader_trait_twice_bright
			} 
		}
		else_if = {
			limit = { 
				from = {
					species = { has_trait = btr_yahg_trait_core }
				}
			}
			from = {
				add_trait = btr_leader_trait_hype_perecpt
			} 
		}

	}
}

# Asari Generals get the Commando trait
country_event = {
	id = btr_asrep.04
	is_triggered_only = yes
	hide_window = yes

	trigger = {
			from = {
				species = { has_trait = mec_asari_trait_core }
			}
	}

	immediate = {
		from = {
			if = {
				limit = { leader_class = general }
				add_trait = btr_leader_trait_commando
			}
		}
	}
}
