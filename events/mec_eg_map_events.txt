# This ONLY works correctly if there's one instance of the same initialiser in the game. Spawning duplicates is not recommended.

# Argos Rho
event = {
	id = mec_eg_map.101
	hide_window = yes
	is_triggered_only = yes
	fire_only_once = yes

	immediate = {
		no_scope = {	
			random_system = {
				limit = { has_star_flag = mec_argos_rho_hydra_system }
				save_global_event_target_as = mec_argos_rho_hydra_system
			}
			random_system = {
				limit = { has_star_flag = mec_argos_rho_gorgon_system }
				save_global_event_target_as = mec_argos_rho_gorgon_system
			}
			random_system = {
				limit = { has_star_flag = mec_argos_rho_phoenix_system }
				save_global_event_target_as = mec_argos_rho_phoenix_system
			}
			
			event_target:mec_argos_rho_hydra_system = {
				add_hyperlane = {
					from = event_target:mec_argos_rho_hydra_system
					to = event_target:mec_argos_rho_gorgon_system
				}
				add_hyperlane = {
					from = event_target:mec_argos_rho_hydra_system
					to = event_target:mec_argos_rho_phoenix_system
				}
			}
		}
	}
}
